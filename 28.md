import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd
df3 = pd.read_csv('day28_groupby.csv')
segment_summary = df3.groupby("segment")[["sales"]].agg(
    ["mean", "median", "std"])
print(segment_summary)

s = df3.groupby("segment")["sales"].agg("mean")
sns.barplot(x=s.index,y=s.values)
plt.title('Sales by segment')
plt.show()
